## Usage

```tsx
import * as React from 'react';
import * as SwitchPrimitive from '@rnr-method/switch';

export default function SwitchScreen() {
  const [checked, setChecked] = React.useState(false);

  return (
    <SwitchPrimitive.Root checked={checked} onCheckedChange={setChecked}>
      <SwitchPrimitive.Thumb />
    </SwitchPrimitive.Root>
  );
}

```

## Props

### Root

Extends [`Pressable`](https://reactnative.dev/docs/pressable#props) props

|     Prop     |           Type           |     Note     |
| :----------: | :----------------------: | :----------: |
|     checked     |   boolean         |  |
|     onCheckedChange     |   (checked: boolean) => void         |  |
|   disabled   |         boolean          | _(optional)_ |
| asChild | boolean | _(optional)_ |
| onKeyDown | (ev: React.KeyboardEvent) => void | Web Only _(optional)_ |

### Thumb

Extends [`View`](https://reactnative.dev/docs/view#props) props

|  Prop   |  Type   |     Note     |
| :-----: | :-----: | :----------: |
| asChild | boolean | _(optional)_ |
